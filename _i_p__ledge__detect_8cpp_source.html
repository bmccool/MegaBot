443 void draw_box_on_foreground(Mat & input_mat, points_t points)
444 {
445  // The plan for this function is to cycle through the points once. We
446  // will maintain a list of rectangles and single points. For each point,
447  // we pick one of the options below, starting with the first, and advancing
448  // down the line if that step is not appropriate.
449  // 1. If a point is close enough to another rectangle, we check to see if
450  // the point is inside the rectangle. If it is inside the rectangle,
451  // we are done with this point. If it is not inside the rectangle, we
452  // expand the rectangle to include it in both the x and y directions if
453  // necessary.
454  // 2. If a point is close enough to another point, we
455  // a. draw a rectangle using the two points as opposite corners
456  // b. remove the first point from the "Single Points" list.
457  // c. add the rectangle to the "Rectangles" list.
458  // 3. If a point is not close enough to another point or a rectangle,
459  // we put it in the "Single Points" list.
460 
461  // What is the maximum distance a pixel can be from another pixel
462  // before it is considered a different object?
463  #define CLOSE_ENOUGH 20 // I'm calling it max_distance
464 
465  boxes_t boxes;
466  int boxes_index = 0;
467  points_t single_points;
468  int single_points_index = 0;
469 
470  bool point_done = false;
471  for (unsigned int i = 0; i < points.size(); i++)
472  {
473  // For each point in the input points_t...
474  // Check to see if this point is close enough to another rectangle
475  for (int j = 0; (j < boxes_index) && (!point_done); j++)

523 bool point_is_close_to_box(Point point_val, Rect box, int close)
524 {
525  // There are two possible cases we can encounter here.
526  // The point can be inside (or on) the box, or the point can
527  // be in any of the eight directions away from the box (think #).
528 
529  // First we should check to see if the point is inside the box.
530  // We can do this by getting the four bounds of the box and checking
531  // if the point is less than the max bound and greater than the min
532  // bound.
533 
534  // Rect holds x,y referring to the top left corner, and height
535  // and width fields. With the above in mind, these are the bounds.
536  // NOTE: Origin is in the top left, x increases to the right, y increases
537  // to the bottom.
538  int left_bound = box.x;
539  int right_bound = box.x + box.width;
540  int top_bound = box.y;
541  int bottom_bound = box.y + box.height;
542 
543  // Next we need to check if we are close to the box.
544  // We need to find where the point is in relationship to the box and its
545  // corners before we can figure this out.
546 
547  if (point_val.x < left_bound)
548  {
549  // We are on the left side of the box
550  if (point_val.y > bottom_bound)
551  {
552  // We are in the bottom left corner.
553  if (distance_between(point_val, Point(left_bound, bottom_bound)) < close) return true;
554  }
555  else if (point_val.y >= top_bound)
556  {
557  // We are on the left side.
558  if (distance_between(point_val, Point(left_bound, point_val.y)) < close) return true;
559  }
560  else
561  {
562  // We are in the top left corner.
563  if (distance_between(point_val, Point(left_bound, top_bound)) < close) return true;
564  }
565  }
566  else if (point_val.x <= right_bound)
567  {
568  // We are in the box, on top of the box, or below the box.
569  // Find out if we are in the box, or to the left or right
570  if (point_val.y > bottom_bound)
571  {
572  // We are below the box.
573  if (distance_between(point_val, Point(point_val.x, bottom_bound)) < close) return true;
574  }
575  else if (point_val.y >= top_bound)
576  {
577  // We are in the box.
578  return true;
579  }
580  else
581  {
582  // We are above the box.
583  if (distance_between(point_val, Point(point_val.x, top_bound)) < close) return true;
584  }
585  }
586  else
587  {
588  // We are on the right side of the box.
589  if (point_val.y > bottom_bound)
590  {
591  // We are in the borrom right corner.
592  if (distance_between(point_val, Point(right_bound, bottom_bound)) < close) return true;
593  }
594  else if (point_val.y >= top_bound)
595  {
596  // We are on the right side.
597  if (distance_between(point_val, Point(right_bound, point_val.y)) < close) return true;
598  }
599  else
600  {
601  // We are in the top right corner.
602  if (distance_between(point_val, Point(right_bound, top_bound)) < close) return true;
603  }
604  }
605  return false;
606 }
